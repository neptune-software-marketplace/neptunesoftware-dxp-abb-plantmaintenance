* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_GET~GET_ALL_FLOC_EQUI
* +-------------------------------------------------------------------------------------------------+
* | [<---] MT_HIER                        TYPE        TY_T_HIER
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_get~get_all_floc_equi.
    DATA: ls_iflot   TYPE iflot,
          lt_hier    LIKE mt_element,
          ls_hier    LIKE LINE OF lt_hier,
          ls_hierout TYPE ty_hier,
          lv_funcloc TYPE iflot-tplnr,
          lv_index   TYPE sy-tabix.
    FIELD-SYMBOLS <hier> LIKE ls_hier.

    FREE: mt_hier[].
* select single * from iflot into ls_iflot where tplma = ''.
*  lv_funcloc = '000'.
*    CALL METHOD ZIF_UI_PM_NOTIF_DXP_TMP_GET=>get_structure
    CALL METHOD ZIF_UI_PM_NOTIF_DXP_TMP_GET~GET_STRUCTURE
      EXPORTING
        i_funcloc   = ms_topnode-tplnr
        i_equipment = abap_true
      RECEIVING
        r_structure = lt_hier.

    SORT lt_hier BY object predecessor.
    DELETE ADJACENT DUPLICATES FROM lt_hier COMPARING object predecessor.


    LOOP AT lt_hier INTO ls_hier.


      MOVE-CORRESPONDING ls_hier TO ls_hierout.

      IF ls_hier-object_type = 'E'.
        ls_hierout-icon         = 'sap-icon://machine'.
        ls_hierout-state        = 'Warning'.
        ls_hierout-histicon     = 'sap-icon://activity-items'.
        ls_hierout-histiconshow = 'X'.
        SHIFT ls_hierout-object LEFT DELETING LEADING '0'.
      ELSE.
        ls_hierout-icon         = 'sap-icon://functional-location'.
        ls_hierout-state        = 'Success'.
        ls_hierout-histicon     = 'sap-icon://activity-items'. " Now functional locations also have history
        ls_hierout-histiconshow = 'X'.
      ENDIF.

      CONCATENATE 'K' ls_hier-object INTO ls_hierout-key.
      IF ls_hier-predecessor IS NOT INITIAL.
        CONCATENATE 'K' ls_hier-predecessor INTO ls_hierout-parent.
      ENDIF.

      CASE  ls_hier-object_type.
        WHEN 'E'.
          SELECT SINGLE eqktx INTO ls_hierout-description
            FROM eqkt WHERE equnr = ls_hier-object
                   AND ( spras = sy-langu OR spras = 'DA' ).
        WHEN 'T'.
          SELECT SINGLE pltxt INTO ls_hierout-description
          FROM iflotx WHERE tplnr = ls_hier-object
          AND ( spras = sy-langu OR spras = 'DA' ).
      ENDCASE.

      CONCATENATE ls_hierout-object ls_hierout-description INTO ls_hierout-searchfield.
      TRANSLATE ls_hierout-searchfield TO UPPER CASE.
      APPEND ls_hierout TO mt_hier.


      CLEAR: ls_hier, ls_hierout.
    ENDLOOP.

    SORT mt_hier BY object.
  ENDMETHOD.


* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_GET~GET_MY_NOTIFICATIONS
* +-------------------------------------------------------------------------------------------------+
* | [<---] MT_NOTIFICATIONS               TYPE        TY_T_NOTIFICATION
* | [<---] MT_NOTES                       TYPE        TY_T_NOTES
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_get~get_my_notifications.
    DATA: lt_qmel         TYPE TABLE OF qmel,
          ls_notification TYPE ty_notification,
          lv_header       TYPE bapi2080_nothdre,
          lv_hdr_txt      TYPE bapi2080_nothdtxte,
          lt_status       TYPE TABLE OF dd07v,
          lt_longtext     TYPE TABLE OF tline,
          lv_stxh         TYPE stxh.

*Notes retrieval variables
    DATA: ls_sofc       TYPE v_sofc,
          ls_note       TYPE ty_attachment,
          lv_doc_id     TYPE sofolenti1-doc_id,
          lt_srgbtbrel  TYPE STANDARD TABLE OF srgbtbrel,
          ls_sofolentil TYPE sofolenti1,
          lv_instid     TYPE string,
          lv_typeid     TYPE string,
          lt_content    TYPE STANDARD TABLE OF solisti1,
          lt_notes      TYPE STANDARD TABLE OF ty_attachment,
          ls_user       TYPE bapiaddr3,
          lt_return     TYPE STANDARD TABLE OF bapiret2.

    DATA: BEGIN OF lv_key,
            foltp TYPE so_fol_tp,
            folyr TYPE so_fol_yr,
            folno TYPE so_fol_no,
            doctp TYPE so_doc_tp,
            docyr TYPE so_doc_yr,
            docno TYPE so_doc_no,
          END OF lv_key.

    TYPES:
      BEGIN OF ty_instid_a,
        instid_a TYPE sibfboriid,
      END OF ty_instid_a.

    DATA: lt_instid_a TYPE STANDARD TABLE OF ty_instid_a,
          lv_instid_a TYPE ty_instid_a.

*Get Notifications
    SELECT qmnum phase FROM viqmel
      INTO CORRESPONDING FIELDS OF TABLE lt_qmel
      WHERE ernam EQ sy-uname
      AND phase NOT IN (4, 5).

    CHECK lt_qmel IS NOT INITIAL.

* Do this so that we can FOR ALL ENTRIES later.
    LOOP AT lt_qmel ASSIGNING FIELD-SYMBOL(<qmel>).
      lv_instid_a-instid_a = <qmel>-qmnum.
      APPEND lv_instid_a TO lt_instid_a.
      CLEAR lv_instid_a.
    ENDLOOP.

*Get Status descriptions
    CALL FUNCTION 'DD_DOMVALUES_GET'
      EXPORTING
        domname        = 'QM_PHASE'
        text           = 'X'
        langu          = sy-langu
      TABLES
        dd07v_tab      = lt_status
      EXCEPTIONS
        wrong_textflag = 1
        OTHERS         = 2.

    "$. Region ### GET NOTES for all notifications before looping ###

    lv_typeid = 'BUS2038'.

* Get GOS Relation
    SELECT *
      FROM srgbtbrel
      INTO CORRESPONDING FIELDS OF TABLE lt_srgbtbrel
      FOR ALL ENTRIES IN lt_instid_a
           WHERE instid_a = lt_instid_a-instid_a
             AND typeid_a EQ lv_typeid
             AND reltype  EQ 'NOTE'
      .

    SORT lt_srgbtbrel BY instid_b.
    DELETE ADJACENT DUPLICATES FROM lt_srgbtbrel COMPARING instid_b.

* Get Attachment/Comment
    LOOP AT lt_srgbtbrel ASSIGNING FIELD-SYMBOL(<ls_srgbtbrel>).

      lv_key = <ls_srgbtbrel>-instid_b.

*   Get Document Info
      SELECT SINGLE *
             FROM v_sofc
             INTO ls_sofc
             WHERE foltp EQ lv_key-foltp
               AND folyr EQ lv_key-folyr
               AND folno EQ lv_key-folno
               AND doctp EQ lv_key-doctp
               AND docyr EQ lv_key-docyr
               AND docno EQ lv_key-docno.

      CHECK sy-subrc EQ 0.

      TRANSLATE ls_sofc-file_ext TO UPPER CASE.
      IF ls_sofc-file_ext EQ 'TXT'.
        ls_sofc-file_ext = 'TXT'.
      ENDIF.

      lv_doc_id = <ls_srgbtbrel>-instid_b.

      CALL FUNCTION 'SO_DOCUMENT_READ_API1'
        EXPORTING
          document_id                = lv_doc_id
        IMPORTING
          document_data              = ls_sofolentil
        TABLES
          object_content             = lt_content
        EXCEPTIONS
          document_id_not_exist      = 1
          operation_no_authorization = 2
          x_error                    = 3
          OTHERS                     = 4.

*   Build String
      LOOP AT lt_content ASSIGNING FIELD-SYMBOL(<content>).
        CONCATENATE ls_note-content <content>-line INTO ls_note-content SEPARATED BY cl_abap_char_utilities=>cr_lf.
      ENDLOOP.

      ls_note-qmnum = <ls_srgbtbrel>-instid_a.
      ls_note-instid_a = <ls_srgbtbrel>-instid_a.
      ls_note-instid_b = <ls_srgbtbrel>-instid_b.
      ls_note-docdes = ls_sofc-docdes.
      ls_note-file_ext = ls_sofc-file_ext.
      ls_note-crdat = ls_sofc-crdat.
      ls_note-crtim = ls_sofc-crtim.
      ls_note-username = ls_sofolentil-creat_name.

      CALL FUNCTION 'BAPI_USER_GET_DETAIL'
        EXPORTING
          username      = ls_note-username
          cache_results = 'X'
        IMPORTING
          address       = ls_user
        TABLES
          return        = lt_return.

      ls_note-user = ls_user-fullname.

      IF ls_note-username EQ sy-uname.
        ls_note-icon = 'sap-icon://fa-regular/user'.
      ELSE.
        ls_note-icon = 'sap-icon://fa-solid/user-shield'.
      ENDIF.

      APPEND ls_note TO mt_notes.
      CLEAR ls_note.

    ENDLOOP.

    SORT mt_notes BY crdat DESCENDING crtim DESCENDING.
    "$. Endregion ### GET NOTES ###

    CLEAR ls_note.

    LOOP AT lt_qmel ASSIGNING FIELD-SYMBOL(<notification>).
      CLEAR ls_notification.

      CALL FUNCTION 'BAPI_ALM_NOTIF_GET_DETAIL'
        EXPORTING
          number             = <notification>-qmnum
        IMPORTING
          notifheader_export = lv_header
          notifhdtext        = lv_hdr_txt.

*Only add it to returning structure if not marked for deletion
      IF NOT lv_header-sys_status CS 'DLFL'.

        MOVE-CORRESPONDING lv_header TO ls_notification.

*   Get Text Header
        SELECT SINGLE *
               FROM stxh
               INTO CORRESPONDING FIELDS OF lv_stxh
               WHERE tdobject EQ 'QMEL'
                 AND tdname	  EQ <notification>-qmnum
                 AND tdspras  EQ sy-langu.

        IF sy-subrc EQ 0.

*     Get Text
          CALL FUNCTION 'READ_TEXT'
            EXPORTING
              id                      = lv_stxh-tdid
              language                = lv_stxh-tdspras
              name                    = lv_stxh-tdname
              object                  = lv_stxh-tdobject
            TABLES
              lines                   = lt_longtext
            EXCEPTIONS
              id                      = 1
              language                = 2
              name                    = 3
              not_found               = 4
              object                  = 5
              reference_check         = 6
              wrong_access_to_archive = 7
              OTHERS                  = 8.
        ENDIF.

*Add appropriate carriage returns
        LOOP AT lt_longtext ASSIGNING FIELD-SYMBOL(<textline>).
          SHIFT <textline>-tdline LEFT DELETING LEADING '*'.
          SHIFT <textline>-tdline LEFT DELETING LEADING space.
          CONCATENATE ls_notification-long_text
                      <textline>-tdline
                      cl_abap_char_utilities=>cr_lf
                      INTO ls_notification-long_text.
        ENDLOOP.

*Are there new messages?
*        SELECT qmnum crdat crtim user username FROM mt_notes AS notes
*          INTO CORRESPONDING FIELDS OF TABLE lt_notes
*          WHERE qmnum EQ <notification>-qmnum.
        REFRESH lt_notes.

        LOOP AT mt_notes INTO ls_note WHERE qmnum EQ <notification>-qmnum.
          APPEND ls_note TO lt_notes.
        ENDLOOP.

        SORT lt_notes BY crdat DESCENDING crtim DESCENDING.

        CLEAR ls_note.
        READ TABLE lt_notes INTO ls_note INDEX 1.

        IF ls_note IS NOT INITIAL AND ( ls_note-username IS NOT INITIAL AND ls_note-username NE sy-uname ).
          ls_notification-icon = 'sap-icon://notification-2'.
          ls_notification-highlight = 'Information'.
        ELSE.
          ls_notification-icon = 'sap-icon://notification'.
          ls_notification-highlight = 'None'.
        ENDIF.

*Other data
        ls_notification-qmnum = <notification>-qmnum.
        ls_notification-equip_desc = lv_hdr_txt-equidescr.
        ls_notification-floc_desc = lv_hdr_txt-funcldescr.
        ls_notification-phase = <notification>-phase.

        IF lv_header-breakdown EQ abap_true.
          ls_notification-brkdown = abap_true.
        ELSE.
          ls_notification-malfunction = abap_true.
        ENDIF.

        READ TABLE lt_status WITH KEY domvalue_l = ls_notification-phase ASSIGNING FIELD-SYMBOL(<status>).
        ls_notification-status_text = <status>-ddtext.

        APPEND ls_notification TO mt_notifications.
      ENDIF.
    ENDLOOP.

    SORT mt_notifications BY qmnum.
  ENDMETHOD.


* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_GET~GET_STRUCTURE
* +-------------------------------------------------------------------------------------------------+
* | [--->] I_FUNCLOC                      TYPE        IFLOT-TPLNR
* | [--->] I_EQUIPMENT                    TYPE        DDBOOL_D
* | [<-()] R_STRUCTURE                    LIKE        MT_ELEMENT
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_get~get_structure.
    DATA: ls_structure TYPE ty_element,
          ls_iflo      TYPE iflo,
          lt_iflo      LIKE TABLE OF ls_iflo,
          lt_iflo_del  LIKE TABLE OF ls_iflo,
          ls_equi      TYPE equi
          .
    DATA: lv_level TYPE string.
    FIELD-SYMBOLS: <struc>    LIKE ls_iflo,
                   <strucout> LIKE ls_structure.
    IF i_funcloc IS NOT INITIAL.
      SELECT SINGLE tplnr pltxt objnr tplma iloan FROM iflo INTO CORRESPONDING FIELDS OF ls_iflo
        WHERE tplnr = i_funcloc
        AND lvorm = ''
        AND spras = sy-langu.
*      and iwerk = '1710'.
      APPEND ls_iflo TO lt_iflo.
    ELSE.
      SELECT  tplnr pltxt objnr tplma iloan FROM iflo INTO CORRESPONDING FIELDS OF TABLE lt_iflo
       WHERE tplma = ''
       AND lvorm = ''
       AND spras = sy-langu.
*      and iwerk = '1710'.
    ENDIF.

    FREE: lt_iflo_del[].
    LOOP AT lt_iflo ASSIGNING <struc>.
      SELECT SINGLE COUNT(*) FROM jest WHERE objnr = <struc>-objnr
        AND ( ( stat = 'I0076' ) OR ( stat = 'I0320' ) )
        AND inact = ''.
      IF sy-subrc EQ 0.
        APPEND <struc> TO lt_iflo_del.
      ENDIF.
    ENDLOOP.
    IF ( NOT lt_iflo_del[] IS INITIAL ).
      "
      " Eliminates invalid functional locations
      SORT lt_iflo BY tplnr.
      SORT lt_iflo_del BY tplnr.
      DELETE ADJACENT DUPLICATES FROM lt_iflo     COMPARING tplnr.
      DELETE ADJACENT DUPLICATES FROM lt_iflo_del COMPARING tplnr.
      LOOP AT lt_iflo_del ASSIGNING <struc>.
        LOOP AT lt_iflo TRANSPORTING NO FIELDS
          WHERE tplnr EQ <struc>-tplnr.
          EXIT. " There are records to eliminate
        ENDLOOP.
        IF ( sy-subrc EQ 0 ).
          DELETE lt_iflo WHERE tplnr EQ <struc>-tplnr.
        ENDIF.
      ENDLOOP.
    ENDIF.

    FREE: lt_iflo_del[].
    LOOP AT lt_iflo ASSIGNING <struc>.
      SELECT tplnr pltxt objnr tplma iloan FROM iflo INTO (ls_iflo-tplnr, ls_iflo-pltxt, ls_iflo-objnr, ls_iflo-tplma, ls_iflo-iloan)
         WHERE tplma = <struc>-tplnr
         AND spras = sy-langu.
*      and iwerk = '1710'.
        SELECT SINGLE COUNT(*) FROM jest WHERE objnr = ls_iflo-objnr
          AND ( ( stat = 'I0076' ) OR ( stat = 'I0320' ) )
          AND inact = ''.
        IF sy-subrc EQ 0.
          APPEND ls_iflo TO lt_iflo_del.
        ELSE.
          APPEND ls_iflo TO lt_iflo.
        ENDIF.
        CLEAR ls_iflo.
      ENDSELECT.
    ENDLOOP.
    IF ( NOT lt_iflo_del IS INITIAL ).
      "
      " Eliminates invalid functional locations
      SORT lt_iflo BY tplnr.
      SORT lt_iflo_del BY tplnr.
      DELETE ADJACENT DUPLICATES FROM lt_iflo     COMPARING tplnr.
      DELETE ADJACENT DUPLICATES FROM lt_iflo_del COMPARING tplnr.
      LOOP AT lt_iflo_del ASSIGNING <struc>.
        LOOP AT lt_iflo TRANSPORTING NO FIELDS
          WHERE tplnr EQ <struc>-tplnr.
          EXIT. " There are records to eliminate
        ENDLOOP.
        IF ( sy-subrc EQ 0 ).
          DELETE lt_iflo WHERE tplnr EQ <struc>-tplnr.
        ENDIF.
      ENDLOOP.
    ENDIF.

    LOOP AT lt_iflo ASSIGNING <struc>.
      APPEND INITIAL LINE TO r_structure ASSIGNING <strucout>.
      MOVE-CORRESPONDING <struc> TO <strucout>.

      CALL FUNCTION 'CONVERSION_EXIT_TPLNR_OUTPUT'
        EXPORTING
          input  = <struc>-tplnr
        IMPORTING
          output = <strucout>-object.
      CALL FUNCTION 'CONVERSION_EXIT_TPLNR_OUTPUT'
        EXPORTING
          input  = <struc>-tplma
        IMPORTING
          output = <strucout>-predecessor.
      <strucout>-description = <struc>-pltxt.
      <strucout>-object_type = 'T'.
    ENDLOOP.


    IF i_equipment IS NOT INITIAL.
      LOOP AT r_structure ASSIGNING <strucout> WHERE object_type = 'T'.
        SELECT c~equnr d~objnr INTO (ls_structure-object, ls_equi-objnr)
          FROM iflot AS a
          JOIN  iloa AS b
          ON a~tplnr = b~tplnr
          JOIN equz AS c
          ON b~iloan = c~iloan
          JOIN equi AS d
          ON c~equnr = d~equnr
          WHERE a~tplnr = <strucout>-tplnr
          AND b~owner = ''
          AND d~lvorm = ''
          AND a~iwerk
          = '1710'.
          ls_structure-object_type = 'E'.
          ls_structure-predecessor = <strucout>-object.
          ls_structure-tplma       = <strucout>-tplnr.

          SELECT SINGLE COUNT(*) FROM jest WHERE objnr = ls_equi-objnr
          AND stat = 'I0076'
          AND inact = ''.
          IF sy-subrc NE 0.
            APPEND ls_structure TO r_structure.
          ENDIF.
        ENDSELECT.
      ENDLOOP.
    ENDIF.
  ENDMETHOD.


* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_GET~INITIALIZE
* +-------------------------------------------------------------------------------------------------+
* | [<---] MT_STATUS                      TYPE        TY_T_STATUS
* | [<---] MT_PRIORITIES                  TYPE        TY_T_PRIORITIES
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_get~initialize.
    DATA:      ls_status LIKE LINE OF mt_status.

    CONSTANTS: c_spras TYPE sy-langu VALUE 'EN',
               c_artpr TYPE artpr VALUE 'PM'.

* Retrieve priorities list
    SELECT * FROM t356_t INTO TABLE mt_priorities
      WHERE spras = c_spras AND
            artpr = c_artpr
      ORDER BY priok.

* Retrieve QM status list
    CALL FUNCTION 'DD_DOMVALUES_GET'
      EXPORTING
        domname        = 'QM_PHASE'
        text           = 'X'
        langu          = sy-langu
      TABLES
        dd07v_tab      = mt_status
      EXCEPTIONS
        wrong_textflag = 1
        OTHERS         = 2.

    ls_status-domvalue_l = 0.
    ls_status-ddtext = 'All'.
    INSERT ls_status INTO mt_status INDEX 1.

* Dont bring deleted or completed
    DELETE mt_status WHERE domvalue_l EQ 4 OR domvalue_l EQ 5.
  ENDMETHOD.


* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_POST~ADD_IMAGE
* +-------------------------------------------------------------------------------------------------+
* | [--->] I_QMNUM                        TYPE        STRING
* | [<-->] MS_IMAGE                       TYPE        TY_ATTACHMENT
* | [<-->] MS_NOTE                        TYPE        TY_ATTACHMENT
* | [<-->] LS_MESSAGE                     TYPE        /NEPTUNE/MESSAGE
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_post~add_image.

    DATA: lv_folder_id   TYPE soodk,
          lv_object_id   TYPE soobjinfi1-object_id,
          lv_rolea       TYPE borident,
          lv_roleb       TYPE borident,
          lv_pre         TYPE string,
          lv_data        TYPE string,
          lv_datax       TYPE xstring,
          lv_doc_type    TYPE soodk-objtp,
          lv_doc_data    TYPE sodocchgi1,
          lv_doc_info    TYPE sofolenti1,
          it_solix       TYPE STANDARD TABLE OF solix,
          lv_qmnum       TYPE qmel-qmnum,
*          ls_message     TYPE /neptune/message,
          lv_qmnum_clean TYPE qmel-qmnum.

    CHECK i_qmnum IS NOT INITIAL.

*Convert QMNUM
    CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
      EXPORTING
        input  = i_qmnum
      IMPORTING
        output = lv_qmnum.

* Get Root Folder
    CALL FUNCTION 'SO_FOLDER_ROOT_ID_GET'
      EXPORTING
        owner     = sy-uname
        region    = 'B'
      IMPORTING
        folder_id = lv_folder_id.

    lv_object_id = lv_folder_id.

    SPLIT ms_image-content AT ',' INTO lv_pre lv_data.

    lv_doc_type = 'JPG'.

*   Document Information

    LOOP AT lt_srgbtbrel ASSIGNING FIELD-SYMBOL(<ls_srgbtbrel>).

      lv_key = <ls_srgbtbrel>-instid_b.

*   Get Document Info
      SELECT SINGLE *
             FROM v_sofc
             INTO ls_sofc
             WHERE foltp EQ lv_key-foltp
               AND folyr EQ lv_key-folyr
               AND folno EQ lv_key-folno
               AND doctp EQ lv_key-doctp
               AND docyr EQ lv_key-docyr
               AND docno EQ lv_key-docno.

      CHECK sy-subrc EQ 0.

      TRANSLATE ls_sofc-file_ext TO UPPER CASE.
      CHECK ls_sofc-file_ext EQ 'JPE' OR ls_sofc-file_ext EQ 'JPG'.
      IF ls_sofc-file_ext EQ 'JPE'.
        ls_sofc-file_ext = 'JPG'.
      ENDIF.

      lv_doc_id = <ls_srgbtbrel>-instid_b.

      CALL FUNCTION 'SO_DOCUMENT_READ_API1'
        EXPORTING
          document_id                = lv_doc_id
        IMPORTING
          document_data              = ls_sofolentil
        TABLES
*         object_header              = it_header
          contents_hex               = lt_hex
        EXCEPTIONS
          document_id_not_exist      = 1
          operation_no_authorization = 2
          x_error                    = 3
          OTHERS                     = 4.

*   Build String
      CLEAR lv_xstring.
      LOOP AT lt_hex ASSIGNING FIELD-SYMBOL(<ls_hex>).
        CONCATENATE lv_xstring
                    <ls_hex>-line
                    INTO lv_xstring IN BYTE MODE.
      ENDLOOP.

*   Base64 Encoding
      CALL FUNCTION 'SCMS_BASE64_ENCODE_STR'
        EXPORTING
          input  = lv_xstring
        IMPORTING
          output = ls_image-content.

      CONCATENATE 'data:image/jpeg;base64,' ls_image-content INTO ls_image-content.

      ls_image-instid_a = <ls_srgbtbrel>-instid_a.
      ls_image-instid_b = <ls_srgbtbrel>-instid_b.
      ls_image-docdes = ls_sofc-docdes.
      ls_image-file_ext = ls_sofc-file_ext.
      ls_image-crdat = ls_sofc-crdat.
      ls_image-crtim = ls_sofc-crtim.

      APPEND ls_image TO mt_images.
      CLEAR ls_image.

    ENDLOOP.

  ENDMETHOD.

* <SIGNATURE>---------------------------------------------------------------------------------------+
* | Instance Public Method ZCL_UI_PM_NOTIFICATION_DXP_TMP->ZIF_UI_PM_NOTIF_DXP_TMP_POST~SAVE_TEXT
* +-------------------------------------------------------------------------------------------------+
* | [--->] MS_NOTIFICATION                TYPE        TY_NOTIFICATION
* | [<-->] LS_MESSAGE                     TYPE        /NEPTUNE/MESSAGE
* +--------------------------------------------------------------------------------------</SIGNATURE>
  METHOD zif_ui_pm_notif_dxp_tmp_post~save_text.

    DATA: lt_longtext TYPE STANDARD TABLE OF string,
          ls_longtext TYPE string,
          lt_final    TYPE STANDARD TABLE OF tline,
          ls_final    LIKE LINE OF lt_final,
          lv_thead    TYPE thead,
          lv_date     TYPE string,
          lv_time     TYPE string,
          lv_user     TYPE string.

    IF ms_notification-long_text IS NOT INITIAL.

*Header timestamp line
      CALL FUNCTION 'CONVERSION_EXIT_LDATE_OUTPUT'
        EXPORTING
          input  = sy-datum
        IMPORTING
          output = lv_date.

      CALL FUNCTION 'CONVERSION_EXIT_THHMM_OUTPUT'
        EXPORTING
          input  = sy-uzeit
        IMPORTING
          output = lv_time.

      CONCATENATE '(' sy-uname ')' INTO lv_user.

      CONCATENATE lv_date
                  lv_time
                  lv_user
                  INTO ls_longtext SEPARATED BY space RESPECTING BLANKS.

      SPLIT ms_notification-long_text AT cl_abap_char_utilities=>newline INTO TABLE lt_longtext.

*Text Header
      lv_thead-tdid     = 'LTXT'.
      lv_thead-tdobject = 'QMEL'.
      lv_thead-tdspras  = sy-langu.
      lv_thead-tdname   = ms_notification-qmnum.

*Get existing text if any
      CLEAR lt_final.
      CALL FUNCTION 'READ_TEXT'
        EXPORTING
          id                      = lv_thead-tdid
          language                = sy-langu
          name                    = lv_thead-tdname
          object                  = lv_thead-tdobject
        TABLES
          lines                   = lt_final
        EXCEPTIONS
          id                      = 1
          language                = 2
          name                    = 3
          not_found               = 4
          object                  = 5
          reference_check         = 6
          wrong_access_to_archive = 7
          OTHERS                  = 8.

*Concatenate existing text with new text
      ls_final-tdformat = '*'.
      ls_final-tdline   = ls_longtext.
      APPEND ls_final TO lt_final.
      CLEAR:  ls_final, ls_longtext.

      LOOP AT lt_longtext ASSIGNING FIELD-SYMBOL(<text>).
        ls_final-tdformat = '*'.
        ls_final-tdline   = <text>.
        APPEND ls_final TO lt_final.
        CLEAR  ls_final.
      ENDLOOP.

*Save Text
      CALL FUNCTION 'SAVE_TEXT'
        EXPORTING
          header          = lv_thead
          savemode_direct = 'X'
        TABLES
          lines           = lt_final
        EXCEPTIONS
          id              = 1
          language        = 2
          name            = 3
          object          = 4
          OTHERS          = 5.

*Update QMEL table Long Text flag so that IW22/IW23 displays the text
      UPDATE qmel SET indtx = abap_true WHERE qmnum = ms_notification-qmnum.

    ENDIF.

  ENDMETHOD.
